.Collection's Key Fields Schema
----
schema:
  $ref: '#/components/schemas/CollectionKeysResponseObject'

CollectionKeysResponseObject:
  required:
  - keys
  - links
  type: object
  properties:
    links:
      type: array
      items:
        $ref: '#/components/schemas/Link'
    keys:
      type: array
      items:
        $ref: '#/components/schemas/CollectionKeysObject'

CollectionKeysObject:
  required:
  - id
  - isDefault
  type: object
  properties:
    isDefault:
      type: boolean
    language:
      type: string
    id:
      type: string
